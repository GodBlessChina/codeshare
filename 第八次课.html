# 分形树
import turtle

def tree(branchLen, t):# (树干长度，turtle对象)
    if branchLen > 5:# 递归结束条件是树干长度小于5
        t.forward(branchLen)# 绘制树干
        t.right(20)# 右倾斜20度
        tree(branchLen-15, t)# 递归调用，绘制右边的小树，树干长度-15
        t.left(40)# 向左回40度，即左倾斜20度
        tree(branchLen-10, t)# 递归调用，绘制左边的小树，树干长度-10
        t.right(20)# 向右回20度，即回正
        t.backward(branchLen)# 沿树干退回原位置

# 谢尔平斯基三角形
import turtle

def drawTriangle(points, color, myTurtle):
    myTurtle.fillcolor(color)# 设置三角形内要填充的颜色
    myTurtle.up()# 画笔抬起
    myTurtle.goto(points[0])# 移动到三角形左顶点
    myTurtle.down()# 画笔落下
    myTurtle.begin_fill()# 开始填充颜色
    myTurtle.goto(points[1])
    myTurtle.goto(points[2])
    myTurtle.goto(points[0])
    myTurtle.end_fill()# 填充结束

def getMid(p1, p2):# 计算并返回中点坐标
    return ((p1[0]+p2[0])/2, (p1[1]+p2[1])/2)

def sierpinski(points, degree, myTurtle):
    #points是一个字典，里面元素用元组表示，存放三角形的三个顶点坐标
    #degree代表要分割的度，即分割三角形的次数
    colormap = ['blue', 'red', 'green', 'white', 'yellow',
                'violet', 'orange']
    drawTriangle(points, colormap[degree], myTurtle)
    if degree > 0:
        sierpinski([points[0], # 对左下角三角形进行递归调用
                        getMid(points[0], points[1]), 
                        getMid(points[0], points[2])], 
                    degree-1, myTurtle)
        sierpinski([points[1], # 对顶部三角形进行递归调用
                        getMid(points[0], points[1]), 
                        getMid(points[1], points[2])], 
                    degree-1, myTurtle)
        sierpinski([points[2], # 对右下角三角形进行递归调用
                       getMid(points[0], points[2]), 
                       getMid(points[1], points[2])], 
                   degree-1, myTurtle)

# 绘制股市图
# 数据.txt 的内容如下：
'''
600004.SH
白云机场
开盘价
6.9500
6.9400
6.9400
6.9400
6.9400
6.8500
6.9400
'''

import turtle

#turtle.screensize(1400, 800, "gray")	#画布像素及颜色
turtle.setup(1400, 800)  #画布大小
turtle.pencolor("black")    #画笔颜色
turtle.pensize(1)  #画笔大小
turtle.speed(4) #画笔速度

#打印横纵坐标
turtle.penup()  #抬起笔，不绘图
turtle.goto(-600, 300)
turtle.pendown()    #落下笔，开始绘图
turtle.goto(-600, -300)
turtle.goto(600, -300)

#打印横坐标间隔
x = -600
y = -300
num = 0
while x < 600:
    turtle.penup()
    turtle.goto(x, y + 10)
    turtle.pendown()
    turtle.goto(x, y)
    turtle.penup()
    turtle.goto(x, y - 20)
    turtle.pendown()
    turtle.write(num, font = ("", 8, ""))
    x += 50
    num += 50

#打印“时间间断”
turtle.penup()
turtle.goto(600, -330)
turtle.pendown()
turtle.write("时间间断",  font = ("", 14, ""))

#打印纵坐标间隔
x = -600
y = -270
num = 6
while y < 300:
    #打印线段
    turtle.penup()
    turtle.goto(x + 10, y)
    turtle.pendown()
    turtle.goto(x, y)
    #打印数字
    turtle.penup()
    turtle.goto(x - 25, y)
    turtle.pendown()
    turtle.write(num, font = ("", 8, ""))
    y += 30
    num += 0.5

#打印“开盘价”
turtle.penup()
turtle.goto(-630, 300)
turtle.pendown()
turtle.write("开盘价",  font = ("", 14, ""))

#重置画笔位置
turtle.penup()
turtle.goto(-600, -300)
turtle.pendown()

#读取文件数据
f = open("数据.txt", "r")
txtList = f.readlines()
f.close()
"""
相当于：
with open("白云机场.txt", "r") as f:
	txtList = f.readlines()
注：with的打开方式不需要手动关闭文件。
"""

#打印折线
turtle.pencolor("red")    #画笔颜色
x = -600
for i in range(3, len(txtList)):
    txtList[i] = txtList[i].rstrip(' \n')   #去掉元素末尾的空格及换行符
    value = float(txtList[i])   #将元素转换成浮点型数据
    # 根据权值计算坐标偏移量并打印
    x += 1
    y = -240 + (value - 6) * 30
    turtle.goto(x, y)
